Java考试复习用

2023-05-16 1 min read 
    public static void main(String args[]) {
        System.out.println("输入回文数：");
        Scanner sc = new Scanner(System.in);
        String Read = sc.nextLine();
        char a[] = Read.toCharArray();
        boolean flag = false;
        for(int i = 0, j = a.length - 1; i != j && i <= j; i ++, j --){
            if(a[i] != a[j]) flag = true;
        }
        if(flag) System.out.println("不是回文数");
        else System.out.println("是回文数");
    }
